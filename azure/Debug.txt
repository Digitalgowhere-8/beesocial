# Live stream (like `docker logs -f`)
az containerapp logs show \
  --name beesocial-backend \
  --resource-group DGW-RG-BS \
  --follow

# Frontend logs
az containerapp logs show \
  --name beesocial-frontend \
  --resource-group DGW-RG-BS \
  --follow

# Last 100 lines (like `docker logs --tail 100`)
az containerapp logs show \
  --name beesocial-backend \
  --resource-group DGW-RG-BS \
  --tail 100

# Force-restart the frontend to pull down the newly pushed  image layer
az containerapp revision restart \
  --name beesocial-frontend \
  --resource-group DGW-RG-BS \
  --revision $(az containerapp revision list --name beesocial-frontend --resource-group DGW-RG-BS --query "[0].name" -o tsv)

# Force-restart the backend as well
az containerapp revision restart \
  --name beesocial-backend \
  --resource-group DGW-RG-BS \
  --revision $(az containerapp revision list --name beesocial-backend --resource-group DGW-RG-BS --query "[0].name" -o tsv)
